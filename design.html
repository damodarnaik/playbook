03 / API

THREE LINES IN.
LIVE IN SECONDS.

Pick a language, hit run, watch the round-trip. This sandbox talks to a real edge region — the latency you see is the latency you ship.

REQUEST 
# publish an event to the orders channel
curl https://api.conduit.dev/v1/channels/orders \
  -H "Authorization: Bearer ck_live_••••" \
  -H "Content-Type: application/json" \
  -d '{"event":"order.created","payload":{"id":8112}}'
import { Conduit } from "@conduit/sdk";

const conduit = new Conduit(process.env.CONDUIT_KEY);

await conduit.publish("orders", {
  event: "order.created",
  payload: { id: 8112 },
});
from conduit import Conduit

client = Conduit(os.environ["CONDUIT_KEY"])

await client.publish("orders", {
    "event": "order.created",
    "payload": {"id": 8112},
})
